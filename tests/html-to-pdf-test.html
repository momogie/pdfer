// C# — Generate PDF with WeasyPrint via CLI
public async Task<byte[]> GeneratePdfAsync(string htmlContent)
{
    var tempFile = Path.GetTempFileName() + ".html";
    var outputFile = Path.GetTempFileName() + ".pdf";

    await File.WriteAllTextAsync(tempFile, htmlContent);

    using var proc = Process.Start(new ProcessStartInfo {
        FileName = "weasyprint",
        Arguments = $"{tempFile} {outputFile}",
        RedirectStandardOutput = true,
        UseShellExecute = false
    });
    await proc!.WaitForExitAsync();

    return await File.ReadAllBytesAsync(outputFile);
}
